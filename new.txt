function movie_library() {
  $('.library-box li > a').not('.library-info-box-footer a, .library-info-box h3 a').click(function(){
    if ( $(window).width() >= 1024 ) {
      $('.library-box li a').removeAttr('data-toggle');

      if( $(this).parent('li').find('.library-info-box').is(':visible') ) {
        $('.library-box li').removeClass('active');
        $('.library-box li .library-info-box').stop(true, true).fadeOut();
      } else {
        $('.library-box li').removeClass('active');
        $('.library-box li .library-info-box').stop(true, true).fadeOut();
        $(this).parent('li').addClass('active');
        $(this).parent('li').find('.library-info-box').stop(true, true).fadeToggle();
      }
      return false;
    } else {
      $('.library-box li').removeClass('active');
      $('.library-box li a').attr('data-toggle', 'modal');
    }
  });
}
